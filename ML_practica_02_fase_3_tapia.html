def create_model():

  model = Sequential([
      Dense(48, input_dim=X_trainReshapeImageRow.shape[1], activation='relu'),
      Dense(48, activation='relu'),
      Dense(10, activation='softmax')
  ])

  model.compile(loss='sparse_categorical_crossentropy',
              optimizer='RMSprop',
              metrics=['accuracy'])

  return model

Se obtiene la matriz de confusin del modelo

In [ ]:
# Calcular metricas de evaluacion
import numpy as np
import matplotlib.pyplot as plt
from sklearn.metrics import confusion_matrix
import seaborn as sns

def _load_label_names():
    return ['airplane', 'automobile', 'bird', 'cat', 'deer', 'dog', 'frog', 'horse', 'ship', 'truck']

def evaluar_modelo(y_test, y_pred):
  # Calcular la matriz de confusin
  conf_matrix = confusion_matrix(y_test, y_pred)

  # Calcular las mtricas de evaluacin a partir de la matriz de confusin
  tp = np.diag(conf_matrix)
  fp = np.sum(conf_matrix, axis=0) - tp
  fn = np.sum(conf_matrix, axis=1) - tp

  accuracy = np.mean(tp / (tp + fn))
  precision = np.mean(tp / (tp + fp))
  recall = np.mean(tp / (tp + fn))
  f1 = 2 * precision * recall / (precision + recall)

  # Imprimir resultados
  print("Accuracy  Precision  Recall  F1-score")
  print(f"{accuracy:.4f}    {precision:.4f}    {recall:.4f}    {f1:.4f}")

  # Crear el plot de la matriz de confusin
  plt.figure(figsize=(12, 10))
  sns.heatmap(conf_matrix, annot=True, fmt='d', cmap='coolwarm')

  # Aadir etiquetas y ttulo
  plt.xlabel('categoras predecidas')
  plt.ylabel('categoras reales')
  plt.xticks(np.arange(conf_matrix.shape[0])+0.5, labels=_load_label_names(),rotation=45, ha="right")
  plt.yticks(np.arange(conf_matrix.shape[1])+0.5, labels=_load_label_names())
  plt.title('Figura 5: Matriz de confusin modelo optimizado')

  # Mostrar el plot
  plt.show()
