Descripcin del dataset CIFAR10

CIFAR-10 es un conjunto de datos utilizado para el reconocimiento de objetos en imgenes. Es uno de los conjuntos de datos ms ampliamente utilizados por investigadores, ingenieros, practicantes y aficionados a la visin computarizada y machine learning. CIFAR son las siglas del Canadian Institute for Advanced Research (Jess, 2020).

Dimensiones: CIFAR contiene 60000 imgenes

Cada imagen es de 32x32pixeles y tienen 3 canales (RGB).

Hay 50000 imgenes de entrenamiento y 10000 para testing.

Hay 10 categoras, donde cada una est representada por 6000 imgenes.

airplane : 0

import numpy as np
import matplotlib.pyplot as plt


import pickle

# Funcin para indicar el nombre de las categoras
def _load_label_names():
    """
    Load the label names from file
    """
    return ['airplane', 'automobile', 'bird', 'cat', 'deer', 'dog', 'frog', 'horse', 'ship', 'truck']

# Funcin para cargar cada batch de imgenes
def load_cfar10_batch(cifar10_dataset_folder_path, batch_id):
    """
    Load a batch of the dataset
    """
    with open(cifar10_dataset_folder_path + '/data_batch_' + str(batch_id), mode='rb') as file:
        batch = pickle.load(file, encoding='latin1')

    #cada batch 10000 imgenes, cada imagen de 32x32x3
    features = batch['data'].reshape((len(batch['data']), 3, 32, 32)).transpose(0, 2, 3, 1)
    labels = batch['labels']

    return features, labels

# Funcin para visualizar una imagen de un batch especfico del dataset y presentar sus caracersticas ms relevantes
def display_stats(cifar10_dataset_folder_path, batch_id, sample_id):
    """
    Display Stats of the the dataset
    """
    batch_ids = list(range(1, 6))

    if batch_id not in batch_ids:
        print('Batch Id out of Range. Possible Batch Ids: {}'.format(batch_ids))
        return None

    features, labels = load_cfar10_batch(cifar10_dataset_folder_path, batch_id)

    if not (0 <= sample_id < len(features)):
        print('{} samples in batch {}.  {} is out of range.'.format(len(features), batch_id, sample_id))
        return None

    print('\nEstadsticas del batch {}:'.format(batch_id))
    print('Samples (cantidad de imgenes): {}'.format(len(features)))
    print('Cantidad de Etiquetas: {}'.format(dict(zip(*np.unique(labels, return_counts=True)))))
    print('Primeras 20 Etiquetas: {}'.format(labels[:20]))

    sample_image = features[sample_id]
    sample_label = labels[sample_id]
    label_names = _load_label_names()

    print('\nImagen #{}:'.format(sample_id))
    print('Imagen - Valor Min: {} Valor Max: {}'.format(sample_image.min(), sample_image.max()))
    print('Imagen - Dimensiones: {}'.format(sample_image.shape))
    print('Etiqueta - Etiqueta Id: {} Nombre: {}'.format(sample_label, label_names[sample_label]))
    #plt.axis('off')
    plt.title("Figura 1: Imagen de muestra del Dataset")
    plt.imshow(sample_image)

# Calcular metricas de evaluacion
import numpy as np
import matplotlib.pyplot as plt
from sklearn.metrics import confusion_matrix
import seaborn as sns

def evaluar_modelo(y_test, y_pred):
  # Calcular la matriz de confusin
  conf_matrix = confusion_matrix(y_test, y_pred)

  # Calcular las mtricas de evaluacin a partir de la matriz de confusin
  tp = np.diag(conf_matrix)
  fp = np.sum(conf_matrix, axis=0) - tp
  fn = np.sum(conf_matrix, axis=1) - tp

  accuracy = np.mean(tp / (tp + fn))
  precision = np.mean(tp / (tp + fp))
  recall = np.mean(tp / (tp + fn))
  f1 = 2 * precision * recall / (precision + recall)

  # Imprimir resultados
  print("Accuracy  Precision  Recall  F1-score")
  print(f"{accuracy:.4f}    {precision:.4f}    {recall:.4f}    {f1:.4f}")

  # Crear el plot de la matriz de confusin
  plt.figure(figsize=(12, 10))
  sns.heatmap(conf_matrix, annot=True, fmt='d', cmap='coolwarm')

  # Aadir etiquetas y ttulo
  plt.xlabel('categoras predecidas')
  plt.ylabel('categoras reales')
  plt.xticks(np.arange(conf_matrix.shape[0])+0.5, labels=_load_label_names(),rotation=45, ha="right")
  plt.yticks(np.arange(conf_matrix.shape[1])+0.5, labels=_load_label_names())
  plt.title('Figura 3: Matriz de confusin modelo base')

  # Mostrar el plot
  plt.show()

evaluar_modelo(y_test, y_pred)

Conclusiones

La validacin cruzada o "k-fold cross-validation" no siempre garantiza una mejora en el accuracy de un modelo. Podemos decir que la validacin cruzada se utiliza principalmente para evaluar el rendimiento, evitar el "overfiting" y estimar cmo se generaliza ante datos no vistos.

Hay que recalcar que en esta prctica se us un modelo de Red neuronal con el fin de obtener un modelo base y buscar la optimizacin (fine tunning) del mismo por motivos de aprendizaje. Para trabajar con imgenes una red neuronal no es la mejor opcin, lo mejor es usar una red neuronal convolucional que se adapta mejora a las imgenes sin necesidad de convertirlas a una sola dimensin como se hizo para la red neuronal.
